What is JavaScript? ‚ö°

JavaScript is a programming language that makes web pages interactive. It can modify HTML content, change CSS styles, respond to user events, and communicate with servers.

Key Features:

  • Dynamic: Can change content and styles in real-time
  • Event-driven: Responds to user interactions
  • Client-side: Runs in the user's browser
  • Versatile: Also works on servers (Node.js)

JavaScript in HTML:

<!-- Inline JavaScript -->
<button onclick="alert('Hello World!')">Click me</button>

<!-- Internal JavaScript -->
<script>
    console.log('Hello from internal script!');
    
    function showAlert() {
        alert('This is a JavaScript function!');
    }
</script>

<!-- External JavaScript -->
<script src="script.js"></script>

Variables üì¶

Variables store data values. JavaScript has three ways to declare variables: var, let, and const.

Variable Declarations:

// Modern way (ES6+) - Use these
let userName = 'John Doe';        // Can be reassigned
const age = 25;                   // Cannot be reassigned
const PI = 3.14159;              // Constants for fixed values

// Older way (avoid in new code)
var oldVariable = 'legacy';      // Function-scoped, can cause issues

// Variable reassignment
let counter = 0;
counter = 1;                     // ‚úÖ OK with let
counter++;                       // ‚úÖ Increment

const name = 'Alice';
// name = 'Bob';                 // ‚ùå Error! Cannot reassign const

// Multiple declarations
let firstName = 'John', 
    lastName = 'Doe', 
    fullName = firstName + ' ' + lastName;

// Destructuring assignment
const person = { name: 'Alice', age: 30 };
const { name: personName, age: personAge } = person;

Variable Naming Rules:

// ‚úÖ Valid variable names
let userName = 'john';
let user_name = 'john';
let $element = document.querySelector('#app');
let _private = 'hidden';
let user1 = 'first user';
let camelCase = 'preferred style';

// ‚ùå Invalid variable names
// let 1user = 'wrong';          // Can't start with number
// let user-name = 'wrong';      // Can't contain hyphens
// let user name = 'wrong';      // Can't contain spaces
// let let = 'wrong';            // Can't use reserved words

// Naming conventions
const MAX_USERS = 100;           // Constants in UPPER_CASE
let isLoggedIn = false;          // Boolean variables with is/has/can
let getUserData = function() {}; // Functions in camelCase
let userName = 'john';           // Variables in camelCase

Data Types üî¢

JavaScript has dynamic typing, meaning variables can hold different types of data. There are primitive and non-primitive data types.

Primitive Data Types:

// Number - integers and decimals
let age = 25;                    // Integer
let price = 99.99;               // Decimal
let infinity = Infinity;         // Special number value
let notANumber = NaN;            // Not a Number

// String - text data
let name = 'John Doe';           // Single quotes
let message = "Hello World!";    // Double quotes
let template = `Hello ${name}`;  // Template literals (backticks)

// Boolean - true or false
let isActive = true;
let isComplete = false;

// Undefined - variable declared but not assigned
let undefinedVariable;
console.log(undefinedVariable);  // undefined

// Null - intentional absence of value
let emptyValue = null;

// Symbol (ES6) - unique identifier
let id = Symbol('id');

// BigInt (ES2020) - large integers
let bigNumber = 9007199254740991n;

Non-Primitive Data Types:

// Object - collection of key-value pairs
let person = {
    name: 'John',
    age: 30,
    city: 'New York',
    isEmployed: true
};

// Array - ordered list of values
let numbers = [1, 2, 3, 4, 5];
let fruits = ['apple', 'banana', 'orange'];
let mixed = [1, 'hello', true, null];

// Function - reusable block of code
function greet(name) {
    return `Hello, ${name}!`;
}

// Date - represents dates and times
let now = new Date();
let birthday = new Date('1990-05-15');

// Regular Expression - pattern matching
let emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

Type Checking and Conversion:

// Type checking with typeof
console.log(typeof 42);          // "number"
console.log(typeof 'hello');     // "string"
console.log(typeof true);        // "boolean"
console.log(typeof undefined);   // "undefined"
console.log(typeof null);        // "object" (known quirk)
console.log(typeof [1, 2, 3]);   // "object"
console.log(typeof {name: 'John'}); // "object"

// Type conversion
let str = '123';
let num = Number(str);           // 123 (number)
let bool = Boolean(str);         // true

let value = 42;
let text = String(value);        // "42" (string)
let automatic = value + '';      // "42" (automatic conversion)

// Parsing strings to numbers
let integer = parseInt('123.45'); // 123
let decimal = parseFloat('123.45'); // 123.45

// Checking for arrays specifically
let arr = [1, 2, 3];
console.log(Array.isArray(arr)); // true

Functions üîß

Functions are reusable blocks of code that perform specific tasks. They can take inputs (parameters) and return outputs.

Function Declaration and Calling:

// Function declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function call
let message = greet('Alice');
console.log(message); // "Hello, Alice!"

// Function with multiple parameters
function addNumbers(a, b) {
    return a + b;
}

let sum = addNumbers(5, 3); // 8

// Function with default parameters
function greetWithDefault(name = 'Guest') {
    return `Welcome, ${name}!`;
}

console.log(greetWithDefault());        // "Welcome, Guest!"
console.log(greetWithDefault('John'));  // "Welcome, John!"

// Function expression
const multiply = function(x, y) {
    return x * y;
};

// Arrow function (ES6)
const divide = (x, y) => {
    return x / y;
};

// Shorter arrow function syntax
const square = x => x * x;
const add = (a, b) => a + b;

Function Examples:

// Function that doesn't return a value
function showAlert(message) {
    alert(message);
    // No return statement = returns undefined
}

// Function with conditional logic
function getGrade(score) {
    if (score >= 90) {
        return 'A';
    } else if (score >= 80) {
        return 'B';
    } else if (score >= 70) {
        return 'C';
    } else if (score >= 60) {
        return 'D';
    } else {
        return 'F';
    }
}

// Function with multiple operations
function processUserData(firstName, lastName, age) {
    // Validate input
    if (!firstName || !lastName || age < 0) {
        return null;
    }
    
    // Process data
    const fullName = firstName + ' ' + lastName;
    const isAdult = age >= 18;
    
    // Return object
    return {
        fullName: fullName,
        age: age,
        isAdult: isAdult,
        canVote: isAdult && age >= 18
    };
}

// Higher-order function (function that takes another function)
function applyOperation(a, b, operation) {
    return operation(a, b);
}

// Usage
const result1 = applyOperation(5, 3, add);      // 8
const result2 = applyOperation(5, 3, multiply); // 15

Control Flow üîÄ

Control flow statements determine the order in which code executes. They include conditionals, loops, and other flow control mechanisms.

Conditional Statements:

// If statement
let age = 18;

if (age >= 18) {
    console.log("You are an adult");
}

// If-else statement
if (age >= 21) {
    console.log("You can drink alcohol");
} else {
    console.log("You cannot drink alcohol");
}

// If-else if-else chain
let score = 85;

if (score >= 90) {
    console.log("Grade A - Excellent!");
} else if (score >= 80) {
    console.log("Grade B - Good job!");
} else if (score >= 70) {
    console.log("Grade C - Average");
} else if (score >= 60) {
    console.log("Grade D - Below average");
} else {
    console.log("Grade F - Failed");
}

// Switch statement
let day = 'Monday';

switch (day) {
    case 'Monday':
        console.log("Start of the work week");
        break;
    case 'Tuesday':
    case 'Wednesday':
    case 'Thursday':
        console.log("Midweek");
        break;
    case 'Friday':
        console.log("TGIF!");
        break;
    case 'Saturday':
    case 'Sunday':
        console.log("Weekend!");
        break;
    default:
        console.log("Invalid day");
}

Loops:

// For loop
for (let i = 0; i < 5; i++) {
    console.log(`Count: ${i}`);
}

// For loop with array
let fruits = ['apple', 'banana', 'orange'];
for (let i = 0; i < fruits.length; i++) {
    console.log(`Fruit ${i + 1}: ${fruits[i]}`);
}

// For...of loop (ES6) - iterates over values
for (let fruit of fruits) {
    console.log(`I like ${fruit}`);
}

// For...in loop - iterates over keys/indices
for (let index in fruits) {
    console.log(`Index ${index}: ${fruits[index]}`);
}

// While loop
let count = 0;
while (count < 3) {
    console.log(`While count: ${count}`);
    count++;
}

// Do-while loop (executes at least once)
let x = 0;
do {
    console.log(`Do-while: ${x}`);
    x++;
} while (x < 3);

// Breaking out of loops
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break; // Exit the loop
    }
    if (i % 2 === 0) {
        continue; // Skip to next iteration
    }
    console.log(`Odd number: ${i}`);
}
