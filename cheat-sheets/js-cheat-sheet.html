Variables

let userName = 'John';     // Can be reassigned

Naming Rules:

camelCase userName, firstName CONSTANTS MAX_USERS, API_URL Valid chars a-z, A-Z, 0-9, _, $ Cannot start Numbers, reserved words

Functions

// Function declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function expression
const add = function(a, b) {
    return a + b;
};

// Arrow function
const multiply = (a, b) => a * b;

// Arrow function with block
const divide = (a, b) => {
    if (b === 0) return 'Error';
    return a / b;
};

// Default parameters
function greetDefault(name = 'Guest') {
    return `Hello, ${name}!`;
}

// Rest parameters
function sum(...numbers) {
    return numbers.reduce((a, b) => a + b, 0);
}

Control Flow

// If statement
if (condition) {
    // code
} else if (otherCondition) {
    // code
} else {
    // code
}

// Ternary operator
const result = condition ? 'true value' : 'false value';

// Switch statement
switch (value) {
    case 'a':
        // code
        break;
    case 'b':
        // code
        break;
    default:
        // code
}

// For loop
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// While loop
while (condition) {
    // code
}

// For...of (values)
for (let item of array) {
    console.log(item);
}

// For...in (keys)
for (let key in object) {
    console.log(key, object[key]);
}

Array Iteration:

// Higher-order methods

arr.map(item => item * 2);
arr.filter(item => item > 1);
arr.reduce((sum, item) => sum + item, 0);
arr.find(item => item > 1);
arr.some(item => item > 1);
arr.every(item => item > 0);

Objects

// Creating objects

    name: 'John',
    age: 30,

// Accessing properties

obj['name'];              // Bracket notation

// Adding/modifying properties
obj.city = 'New York';
obj['country'] = 'USA';

// Deleting properties
delete obj.age;

// Object methods
Object.keys(obj);         // Get keys
Object.values(obj);       // Get values
Object.entries(obj);      // Get key-value pairs
Object.assign({}, obj);   // Shallow copy
{...obj};                 // Spread syntax copy

Error Handling

// Try-catch
try {
    // Code that might throw error
    let result = riskyOperation();
} catch (error) {
    console.error('Error:', error.message);
} finally {
    // Always runs
    console.log('Cleanup');
}

// Throwing errors
throw new Error('Custom error message');
throw new TypeError('Wrong type');

// Promise error handling
fetch('/api/data')
    .then(response => response.json())
    .catch(error => console.error(error));

// Async/await error handling
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Fetch error:', error);
    }
}
